59int
60sys_sleep(void)
61{
62 int n;
63 uint ticks0;
64
65 if(argint(0, &n) < 0)
66 return -1;
67 acquire(&tickslock);
68 ticks0 = ticks;
69 while(ticks - ticks0 < n){
70 if(myproc()->killed){
71 release(&tickslock);
72 return -1;
73 }
74 sleep(&ticks, &tickslock);
75 }
76 release(&tickslock);
77 return 0;
78}
