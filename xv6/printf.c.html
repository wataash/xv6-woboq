41{

45
46 state = 0;
47 ap = (uint*)(void*)&fmt + 1;
48 for(i = 0; fmt[i]; i++){

50 if(state == 0){
51 if(c == '%'){
52 state = '%';
53 } else {
54 putc(fd, c);
55 }
56 } else if(state == '%'){
57 if(c == 'd'){
58 printint(fd, *ap, 10, 1);
59 ap++;
60 } else if(c == 'x' || c == 'p'){
61 printint(fd, *ap, 16, 0);
62 ap++;
63 } else if(c == 's'){
64 s = (char*)*ap;
65 ap++;
66 if(s == 0)
67 s = "(null)";
68 while(*s != 0){
69 putc(fd, *s);
70 s++;
71 }
72 } else if(c == 'c'){
73 putc(fd, *ap);
74 ap++;
75 } else if(c == '%'){
76 putc(fd, c);
77 } else {
78 // Unknown % sequence. Print it to draw attention.
79 putc(fd, '%');
80 putc(fd, c);
81 }
82 state = 0;
83 }
