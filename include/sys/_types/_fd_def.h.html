28#ifndef _FD_SET
29#define _FD_SET
30
31#include <machine/types.h> /* __int32_t */
32
33/*
34 * Select uses bit masks of file descriptors in longs. These macros
35 * manipulate such bit fields (the filesystem macros use chars). The
36 * extra protection here is to permit application redefinition above
37 * the default size.
38 */
39#ifdef FD_SETSIZE
40#define __DARWIN_FD_SETSIZE FD_SETSIZE
41#else /* !FD_SETSIZE */
42#define __DARWIN_FD_SETSIZE 1024
43#endif /* FD_SETSIZE */
44#define __DARWIN_NBBY 8 /* bits in a byte */
45#define __DARWIN_NFDBITS (sizeof(__int32_t) * __DARWIN_NBBY) /* bits per mask */
46#define __DARWIN_howmany(x, y) ((((x) % (y)) == 0) ? ((x) / (y)) : (((x) / (y)) + 1)) /* # y's == x bits? */
47
48__BEGIN_DECLS
49typedef struct fd_set {
50 __int32_t fds_bits[__DARWIN_howmany(__DARWIN_FD_SETSIZE, __DARWIN_NFDBITS)];
51} fd_set;
52__END_DECLS
53
54/* This inline avoids argument side-effect issues with FD_ISSET() */
55static __inline int
56__darwin_fd_isset(int _n, const struct fd_set *_p)
57{
58 return _p->fds_bits[(unsigned long)_n / __DARWIN_NFDBITS] & ((__int32_t)(((unsigned long)1) << ((unsigned long)_n % __DARWIN_NFDBITS)));
59}
60
61#define __DARWIN_FD_SET(n, p) do { int __fd = (n); ((p)->fds_bits[(unsigned long)__fd/__DARWIN_NFDBITS] |= ((__int32_t)(((unsigned long)1)<<((unsigned long)__fd % __DARWIN_NFDBITS)))); } while(0)
62#define __DARWIN_FD_CLR(n, p) do { int __fd = (n); ((p)->fds_bits[(unsigned long)__fd/__DARWIN_NFDBITS] &= ~((__int32_t)(((unsigned long)1)<<((unsigned long)__fd % __DARWIN_NFDBITS)))); } while(0)
63#define __DARWIN_FD_ISSET(n, p) __darwin_fd_isset((n), (p))
64
65#if __GNUC__ > 3 || __GNUC__ == 3 && __GNUC_MINOR__ >= 3
66/*
67 * Use the built-in bzero function instead of the library version so that
68 * we do not pollute the namespace or introduce prototype warnings.
69 */
70#define __DARWIN_FD_ZERO(p) __builtin_bzero(p, sizeof(*(p)))
71#else
72#define __DARWIN_FD_ZERO(p) bzero(p, sizeof(*(p)))
73#endif

75#define __DARWIN_FD_COPY(f, t) bcopy(f, t, sizeof(*(f)))
76#endif /* _FD_SET */
