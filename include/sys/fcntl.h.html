88/*
89 * File status flags: these are used by open(2), fcntl(2).
90 * They are also used (indirectly) in the kernel file structure f_flags,
91 * which is a superset of the open/fcntl flags. Open flags and f_flags
92 * are inter-convertible using OFLAGS(fflags) and FFLAGS(oflags).
93 * Open/fcntl flags begin with O_; kernel-internal flags begin with F.
94 */
95/* open-only flags */
96#define O_RDONLY 0x0000 /* open for reading only */
97#define O_WRONLY 0x0001 /* open for writing only */
98#define O_RDWR 0x0002 /* open for reading and writing */
99#define O_ACCMODE 0x0003 /* mask for above modes */
100
101/*
102 * Kernel encoding of open mode; separate read and write bits that are
103 * independently testable: 1 greater than the above.
104 *
105 * XXX
106 * FREAD and FWRITE are excluded from the #ifdef KERNEL so that TIOCFLUSH,
107 * which was documented to use FREAD/FWRITE, continues to work.
108 */

110#define FREAD 0x0001
111#define FWRITE 0x0002
112#endif
113#define O_NONBLOCK 0x0004 /* no delay */
114#define O_APPEND 0x0008 /* set append mode */

450/*
451 * For F_LOG2PHYS this information is passed back to user
452 * Currently only devoffset is returned - that is the VOP_BMAP
453 * result - the disk device address corresponding to the
454 * current file offset (likely set with an lseek).
455 *
456 * The flags could hold an indication of whether the # of
457 * contiguous bytes reflects the true extent length on disk,
458 * or is an advisory value that indicates there is at least that
459 * many bytes contiguous. For some filesystems it might be too
460 * inefficient to provide anything beyond the advisory value.
461 * Flags and contiguous bytes return values are not yet implemented.
462 * For them the fcntl will nedd to switch from using BMAP to CMAP
463 * and a per filesystem type flag will be needed to interpret the
464 * contiguous bytes count result from CMAP.
465 *
466 * F_LOG2PHYS_EXT is a variant of F_LOG2PHYS that uses a passed in
467 * file offset and length instead of the current file offset.
468 * F_LOG2PHYS_EXT operates on the same structure as F_LOG2PHYS, but
469 * treats it as an in/out.
470 */
471#pragma pack(4)
472
473struct log2phys {
474 unsigned int l2p_flags; /* unused so far */
475 off_t l2p_contigbytes; /* F_LOG2PHYS: unused so far */
476 /* F_LOG2PHYS_EXT: IN: number of bytes to be queried */
477 /* OUT: number of contiguous bytes at this position */
478 off_t l2p_devoffset; /* F_LOG2PHYS: OUT: bytes into device */
479 /* F_LOG2PHYS_EXT: IN: bytes into file */
480 /* OUT: bytes into device */
481};
